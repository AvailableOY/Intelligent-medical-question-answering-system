<script>
	export default {
		// 数据部分（当前为空）
		data() {
			return {
				dt_bgc:"#f7f7f7",
				// 可在此添加输入框内容、消息列表等数据
				msg: "",
				type: 'center',
				chatLists: [

				]
			}
		},
		// 方法部分（当前为空）
		methods: {
			deepThinking(){
				this.ds_bgc = "#000000"
				console.log('背景色已更新为:', this.ds_bgc);
			},
			newChat(){
				this.msg = '',
				this.chatLists = []
			},
			// 可在此添加发送消息、切换模式、上传文件等方法
			sendMsg() {
				let msg = this.msg.trim() // 去除首尾空格
				if (msg === "") {
					// 必须输入对话内容
					uni.showToast({
						title: "主公 请输入",
						duration: 1000
					})
					return
				}
				//数据正在处理
				uni.showLoading({
					title: "别吵 我在思考",
					mask: true
				})
				console.log(this.msg)
				//追加提问到对话列表
				this.chatLists.push({
					role: "user",
					content: msg
				})
				uni.request({
					url: 'http://172.23.241.16:8000/v1/chat/completions',
					data: {
						model: "/data/huggingface/deepseek-1.5B",
						messages: [{
								"role": "system",
								"content": "你是很棒的AI助手"
							},
							{
								"role": "user",
								"content": msg
							}
						],
						"stream": false
					},
					method: 'POST',
					header: {
						'Content-Type': 'application/json' //自定义请求头信息
					},
					success: (response) => {
						let data = response.data.choices[0].message.content;
						// this.text = 'request success';
						let reasoning = data.match(/<think>([\s\S]*?)<\/think>/)?.[1]?.trim() || "";
						let answer = data.replace(/<think>[\s\S]*?<\/think>/, "").trim();

						console.log("推理部分：", reasoning);
						console.log("回答部分：", answer)
						//追加
						this.chatLists.push({
							role: 'assistant',
							content: data
						})
						// 关闭提示框
						uni.hideLoading()
						// 输入框里面的数据需要清空
						this.msg = ''
					}
				});
			},
			change(e) {
				console.log('当前模式：' + e.type + ',状态：' + e.show);
			},
			toggle(type) {
				this.type = type
				// open 方法传入参数 等同在 uni-popup 组件上绑定 type属性
				this.$refs.popup.open(type)
			}
		}
	}
</script>
